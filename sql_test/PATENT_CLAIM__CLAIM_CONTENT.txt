SELECT
	B.APPL_NO
	, B.APPL_DT
	, B.OPN_NO
	, B.OPN_DT
	, B.RGST_NO
	, B.RGST_DT
	, B.PUB_NO
	, B.PUB_DT
	, B.ORG_APPL_NO
	, B.ORG_APPL_DT
	, B.TRNST_SBMT_DT
	, B.INTNL_APPL_NO
	, B.INTNL_APPL_DT
	, B.INTNL_OPN_NO
	, B.INTNL_OPN_DT
	, B.TOFIV
	, B.EN_TOFIV
	, B.LAST_DSPST_CD
	, B.RGST_TPCD
	, B.RFOEX_YN
	, B.RFOEX_DT
	, B.BASIC_IT_CNT
	, B.CHG_DMND_IT_CNT
	, B.APPL_TPCD
	, B.TECH_TRANS_HOPE_FG
	, B.TECH_LD_HOPE_FG
	, I.IPC
	, A.ABSTRACT_CONTENT
	, C.CLAIM_CONTENT
	, S.SPEC_CONTENT
FROM 
	PATENT_BIBLOGRAPHY B,
	(SELECT APPL_NO, UPPER(SUBSTRING(IPC_CD FROM 1 FOR 1)) AS IPC
	FROM PATENT_IPC
	WHERE 1=1
		AND PAT_CLSS_SEQ = '1'
		AND UPPER(SUBSTRING(IPC_CD FROM 1 FOR 1)) = 'A' -- 646908
	) I 
	LEFT JOIN PATENT_ABSTRACT A ON I.APPL_NO = A.APPL_NO
	LEFT JOIN PATENT_CLAIM C ON I.APPL_NO = C.APPL_NO
	LEFT JOIN PATENT_SPECIFICATION S ON I.APPL_NO = S.APPL_NO
	WHERE 
		1=1
		AND I.APPL_NO = B.APPL_NO
		AND B.APPL_DT = '20200102'
	LIMIT 10
;

SELECT
	B.APPL_NO
	, B.APPL_DT
	, B.OPN_NO
	, B.OPN_DT
	, B.RGST_NO
	, B.RGST_DT
	, B.PUB_NO
	, B.PUB_DT
	, B.ORG_APPL_NO
	, B.ORG_APPL_DT
	, B.TRNST_SBMT_DT
	, B.INTNL_APPL_NO
	, B.INTNL_APPL_DT
	, B.INTNL_OPN_NO
	, B.INTNL_OPN_DT
	, B.TOFIV
	, B.EN_TOFIV
	, B.LAST_DSPST_CD
	, B.RGST_TPCD
	, B.RFOEX_YN
	, B.RFOEX_DT
	, B.BASIC_IT_CNT
	, B.CHG_DMND_IT_CNT
	, B.APPL_TPCD
	, B.TECH_TRANS_HOPE_FG
	, B.TECH_LD_HOPE_FG
	, I.IPC
	, A.ABSTRACT_CONTENT
	, C.CLAIM_CONTENT
	, S.SPEC_CONTENT
FROM PATENT_BIBLOGRAPHY B,
	PATENT_ABSTRACT A,
	PATENT_CLAIM C,
	PATENT_SPECIFICATION S,
	(SELECT APPL_NO, UPPER(SUBSTR(IPC_CD,0,1)) AS IPC
	FROM PATENT_IPC
	WHERE 1=1
		AND PAT_CLSS_SEQ = '1'
		AND UPPER(SUBSTR(IPC_CD,0,1)) = 'A'
	) I
	WHERE 1=1
	AND I.APPL_NO = B.APPL_NO
	AND I.APPL_NO = A.APPL_NO(+)
	AND I.APPL_NO = C.APPL_NO(+)
	AND I.APPL_NO = S.APPL_NO(+)
    AND B.APPL_DT = '20200102'
;
